'); $chat.scrollTop($chat[0].scrollHeight); }; chatSocket.onclose = function(e){ console.error('Chat socket closed unexpectedly'); }; //处理发送消息 var $input = $('#chat-message-input'); var $submit = $('#chat-message-submit'); $submit.click(function(){ var message = $input.val(); if(message){ //发送消息 chatSocket.send(JSON.stringify({'message':message})); //清除输入 $input.val(''); //聚焦到输入框 $input.focus(); } }); //一开始就聚焦到输入框上 $input.focus(); $input.keyup(function(e){ if(e.which===13){ //按回车键也可以发消息 $submit.click(); } }) {%endblock%}
